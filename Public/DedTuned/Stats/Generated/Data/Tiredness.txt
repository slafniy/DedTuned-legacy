# PASSIVE: Adds Tiredness Debuffs if condition matched
new entry "TIREDNESS_ADD_DEBUFF"
type "PassiveData"
data "Properties" "IsHidden"
data "StatsFunctorContext" "OnTurn"
data "StatsFunctors" "IF(StatusDurationEqualOrMoreThan(context.Source,'TIREDNESS_DEBUFF_STACKS',2) and not HasStatus('TIREDNESS_DEBUFF_LEVEL_1')):ApplyStatus(SELF,TIREDNESS_DEBUFF_LEVEL_1,100,1)
                     ;IF(StatusDurationEqualOrMoreThan(context.Source,'TIREDNESS_DEBUFF_STACKS',3) and not HasStatus('TIREDNESS_DEBUFF_LEVEL_2')):ApplyStatus(SELF,TIREDNESS_DEBUFF_LEVEL_2,100,1)"
